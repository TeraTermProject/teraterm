Example

ipaddr_size = 10
strdim ipaddr ipaddr_size

getipv4addr ipaddr num
if result = -1 then
    messagebox 'Error' 'IPv4 address'
elseif result = 0 then
    N = ipaddr_size - 1
    for i 0 N
        messagebox ipaddr[i] "IPv4 address"
    next
    N = num - ipaddr_size
    sprintf2 str "And %d address(es)." N
    messagebox str "IPv4 address"
else
    N = num - 1
    for i 0 N
        messagebox ipaddr[i] "IPv4 address"
    next
endif
