include

Moves control to the include file. 

include <include file name>

Example

----- Main file "main.ttl" ------
i=10
:loop
; Move to the include file.
include 'sub.ttl'
if i>=0 goto loop
end
----- End of "main.ttl" ---------

----- Include file "sub.ttl" ----
if i<0 then
  messagebox 'error!' 'sub'
  ; Go back to the main file.
  exit
endif
i = i - 1
; Go back to the main file.
----- End of "sub.ttl" ----------
