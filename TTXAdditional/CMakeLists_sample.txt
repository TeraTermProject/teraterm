  )

if(SUPPORT_OLD_WINDOWS)
  target_link_libraries(
    ${PROJECT_NAME}
    PRIVATE
    layer_for_unicode
    )

    target_sources(
      ${PROJECT_NAME}
      PRIVATE
      ../teraterm/common/compat_w95_vs2005.c

    target_sources(
      ${PROJECT_NAME}

  PRIVATE
